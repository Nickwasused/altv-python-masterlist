 72def request(url: str, server: any = None) -> dict | None:
 73    """This is the common request function to fetch remote data.
 74
 75    Args:
 76        url (str): The Url to fetch.
 77        server (Server): An alt:V masterlist or altstats Server object.
 78
 79    Returns:
 80        None: When an error occurred. But exceptions will still be logged!
 81        json: As data
 82    """
 83    # Use the User-Agent: AltPublicAgent, because some servers protect their CDN with
 84    # a simple User-Agent check e.g. https://luckyv.de does that
 85    with requests.session() as session:
 86        retries = Retry(total=5, backoff_factor=1, status_forcelist=[502, 503, 504])
 87        session.mount('http', HTTPAdapter(max_retries=retries))
 88
 89        if server and "http://" in url and not server.useCdn:
 90            session.headers = RequestHeaders(server).to_dict()
 91        else:
 92            session.headers = {
 93                'User-Agent': Extra.user_agent.value,
 94                'content-type': 'application/json; charset=utf-8'
 95            }
 96
 97        try:
 98            api_request = session.get(url, timeout=5)
 99            if api_request.status_code != 200:
100                logging.warning(f"the request returned nothing.")
101                return None
102            else:
103                return api_request.json()
104        except Exception as e:
105            logging.error(e)
106            return None

245        """Averages - Returns averages data about the specified server (TIME = 1d, 7d, 31d)
246
247        Args:
248            time (str): The timerange of the data. Can be 1d, 7d, 31d.
249            return_result (bool): Define if you want the overall average.

252            None: When an error occurs
253            dict: The maximum player data
254            int: Overall average of defined timerange
255        """
256        average_data = request(MasterlistUrls.specific_server_average.value.format(self.id, time))
257        if not average_data:

261            players_all = 0
262            for entry in average_data:
263                players_all = players_all + entry["c"]
264            result = players_all / len(average_data)
265            return round(result)

267            return average_data

270    def connect_json(self) -> dict | None:
271        """This function fetched the connect.json of an alt:V server.
272
273        Returns:
274            None: When an error occurred. But exceptions will still be logged!
275            dict: The connect.json
276        """
277        if not self.available or self.passworded:
278            return None
279
280        if not self.useCdn:
281            # This Server is not using a CDN.
282            cdn_request = request(f"http://{self.address}/connect.json", self)
283            if cdn_request is None:
284                # possible server error or blocked
285                return None
286            else:
287                return cdn_request
288        else:
289            # let`s try to get the connect.json
290            match self.cdnUrl:
291                case _ if ":80" in self.cdnUrl:
292                    cdn_request = request(f"http://{self.cdnUrl.replace(':80', '')}/connect.json", self)
293                case _ if ":443" in self.cdnUrl:
294                    cdn_request = request(f"https://{self.cdnUrl.replace(':443', '')}/connect.json", self)
295                case _:
296                    cdn_request = request(f"{self.cdnUrl}/connect.json", self)
297
298            if cdn_request is None:
299                # maybe the CDN is offline
300                return None
301            else:
302                return cdn_request

361    def get_dtc_url(self, password=None) -> str | None:
362        """This function gets the direct connect protocol url of an alt:V Server.
363        (https://docs.altv.mp/articles/connectprotocol.html)
364
365        Args:
366            password (str): The password of the server.
367
368        Returns:
369            None: When an error occurred. But exceptions will still be logged!
370            str: The direct connect protocol url.
371        """
372        with StringIO() as dtc_url:
373            if self.useCdn:
374                if "http" not in self.cdnUrl:
375                    dtc_url.write(f"altv://connect/http://{self.cdnUrl}")
376                else:
377                    dtc_url.write(f"altv://connect/{self.cdnUrl}")
378            else:
379                dtc_url.write(f"altv://connect/{self.address}")
380
381            if self.passworded and password is None:
382                logging.warning(
383                    "Your server is password protected but you did not supply a password for the Direct Connect Url.")
384
385            if password is not None:
386                dtc_url.write(f"?password={password}")
387
388            return dtc_url.getvalue()

453def validate_id(server_id: any) -> bool:
454    """Validate a server id
455
456    Args:
457        server_id (any): The id you want to check.
458
459    Returns:
460        bool: True = valid, False = invalid
461    """
462    if not isinstance(server_id, str):
463        return False

466    if result is not None:

468    else:
